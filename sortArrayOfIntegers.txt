#!/bin/bash

echo "Size of array"
read n

arr=()

for((i=0; i<n ;i++));
do
	read arr[$i]
done

echo "Array in normal order: " ${arr[*]}

for((i=0; i<n; i++));
do
	for((j=0; i<n-i-1; j++));
	do
		if [ ${arr[j]} -gt ${arr[$((j+1))]} ];
		then
			temp=${arr[j]}
			arr[$j]=${arr[$((j+1))]}
			arr[$((j+1))]=$temp
		fi
	done
done

echo "Array in sorted order: " ${arr[*]}
